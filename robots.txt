# AmbiSecure — robots policy
# Site is a fully static deploy. The following paths return 410/403 at
# the server level (.htaccess), but listing them here gives well-behaved
# crawlers an earlier signal so they stop wasting budget on probes.

User-agent: *
Allow: /

# Rendering assets must stay crawlable. All CSS/JS/font/image assets live
# under /assets/ and are referenced with a ?v=NN cache-busting query string
# (e.g. /assets/css/main.css?v=32). Without this rule the broad
# "Disallow: /*?" below would block every stylesheet and script from
# Googlebot and break rendering. "/assets/" (longer path) is more specific
# than "/*?", so Google's longest-match rule keeps these Allowed.
Allow: /assets/

# Operator-only dashboards (denied at server level, double-belt here).
Disallow: /_internal/

# WordPress-shaped probe surfaces — site has never run WP; these are
# always spam-network fingerprints.
Disallow: /wp-admin/
Disallow: /wp-includes/
Disallow: /wp-content/
Disallow: /wp-json/
Disallow: /wp-login.php
Disallow: /xmlrpc.php

# Disallow indexing of any query-string variants of canonical pages.
# All real navigation uses clean paths; ?foo=bar URLs are either crawler
# noise or spam-poisoning probes.
Disallow: /*?

Sitemap: https://ambisecure.ambimat.com/sitemap.xml
